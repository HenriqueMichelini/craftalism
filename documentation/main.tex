\chapter{Shared Environment Variables}
\label{app:shared-env}

The following environment variables are referenced by more than one module.
Each module's Configuration section documents the full set of variables for that module;
this appendix highlights variables whose values must be consistent across services.

\rowcolors{2}{CraftLight}{white}
\begin{tabularx}{\linewidth}{@{} L{4.8cm} X @{}}
  \toprule
  {\tableheadstyle Variable} & {\tableheadstyle Consistency requirement} \\
  \midrule
  \code{AUTH\_ISSUER\_URI} &
    Must be set to the same value in the Craftalism API, the Authorization Server,
    and the Minecraft plugin. A mismatch causes JWT validation failures
    because the \code{iss} claim in issued tokens will not match what the API expects. \\
  \code{MINECRAFT\_CLIENT\_SECRET} &
    Must match between the Authorization Server (where the client is seeded)
    and the Minecraft plugin (where it is used to request tokens). \\
  \code{RSA\_PRIVATE\_KEY} / \code{RSA\_PUBLIC\_KEY} &
    Must remain stable across Authorization Server restarts.
    Rotating these keys invalidates all previously issued tokens. \\
  \bottomrule
\end{tabularx}

% Appendix B — Quick-start command reference
\chapter{Quick-Start Command Reference}
\label{app:quickstart}

\subsection*{Start the full platform}
\begin{bashlist}
cd craftalism-deployment
cp env.example .env
# Edit .env with required secrets
docker compose pull
docker compose up -d
\end{bashlist}

\subsection*{Health checks}
\begin{bashlist}
curl -f http://localhost:3000/actuator/health   # Craftalism API
curl -f http://localhost:9000/actuator/health   # Authorization Server
curl -I  http://localhost:8080/                 # Dashboard
\end{bashlist}

\subsection*{Run a module's tests}
\begin{bashlist}
# Java modules (API, Authorization Server)
cd [module]/java && ./gradlew test

# Minecraft plugin
cd craftalism-economy/java && ./gradlew test
\end{bashlist}

\subsection*{Request a token manually}
\begin{bashlist}
curl -X POST 'http://localhost:9000/oauth2/token' \
  -H 'Content-Type: application/x-www-form-urlencoded' \
  -u "minecraft-server:${MINECRAFT_CLIENT_SECRET}" \
  -d 'grant_type=client_credentials&scope=api:read api:write'
\end{bashlist}
